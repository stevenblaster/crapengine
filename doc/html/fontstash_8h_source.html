603 static int fons__atlasRectFits(struct FONSatlas* atlas, int i, int w, int h)
604 {
605  // Checks if there is enough space at the location of skyline span 'i',
606  // and return the max height of all skyline spans under that at that location,
607  // (think tetris block being dropped at that position). Or -1 if no space found.
608  int x = atlas->nodes[i].x;
609  int y = atlas->nodes[i].y;
610  int spaceLeft;
611  if (x + w > atlas->width)
612  return -1;
613  spaceLeft = w;

615  if (i == atlas->nnodes) return -1;
616  y = fons__maxi(y, atlas->nodes[i].y);
617  if (y + h > atlas->height) return -1;
618  spaceLeft -= atlas->nodes[i].width;
619  ++i;
620  }
