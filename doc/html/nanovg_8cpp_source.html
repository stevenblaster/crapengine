358 static float nvg__hue(float h, float m1, float m2)
359 {
360  if (h < 0) h += 1;
361  if (h > 1) h -= 1;
362  if (h < 1.0f/6.0f)
363  return m1 + (m2 - m1) * h * 6.0f;
364  else if (h < 3.0f/6.0f)
365  return m2;
366  else if (h < 4.0f/6.0f)
367  return m1 + (m2 - m1) * (2.0f/3.0f - h) * 6.0f;
368  return m1;
369 }

371 struct NVGcolor nvgHSLA(float h, float s, float l, unsigned char a)
372 {
373  float m1, m2;
374  struct NVGcolor col;
375  h = nvg__modf(h, 1.0f);
376  if (h < 0.0f) h += 1.0f;
377  s = nvg__clampf(s, 0.0f, 1.0f);
378  l = nvg__clampf(l, 0.0f, 1.0f);
379  m2 = l <= 0.5f ? (l * (1 + s)) : (l + s - l * s);
380  m1 = 2 * l - m2;
381  col.r = nvg__clampf(nvg__hue(h + 1.0f/3.0f, m1, m2), 0.0f, 1.0f);
382  col.g = nvg__clampf(nvg__hue(h, m1, m2), 0.0f, 1.0f);
383  col.b = nvg__clampf(nvg__hue(h - 1.0f/3.0f, m1, m2), 0.0f, 1.0f);
384  col.a = a/255.0f;
385  return col;
386 }
