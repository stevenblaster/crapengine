149  static struct GLNVGtexture* glnvg__allocTexture(struct GLNVGcontext* gl)
150  {
151  struct GLNVGtexture* tex = NULL;
152  int i;
153 
154  for (i = 0; i < gl->ntextures; i++)
155  {
156  if (gl->textures[i].id.idx == bgfx::invalidHandle)
157  {
158  tex = &gl->textures[i];
159  break;
160  }
161  }
162 
163  if (tex == NULL)
164  {
165  if (gl->ntextures+1 > gl->ctextures)
166  {
167  int old = gl->ctextures;
168  gl->ctextures = (gl->ctextures == 0) ? 2 : gl->ctextures*2;
169  gl->textures = (struct GLNVGtexture*)realloc(gl->textures, sizeof(struct GLNVGtexture)*gl->ctextures);
170  memset(&gl->textures[old], 0xff, (gl->ctextures-old)*sizeof(struct GLNVGtexture) );
171 
172  if (gl->textures == NULL)
173  {
174  return NULL;
175  }
176  }
177  tex = &gl->textures[gl->ntextures++];
178  }
179 
180  memset(tex, 0, sizeof(*tex));
181

364  static void glnvg__xformInverse(float* inv, float* t)
365  {
366  double invdet, det = (double)t[0] * t[3] - (double)t[2] * t[1];
367  if (det > -1e-6 && det < 1e-6) {
368  glnvg__xformIdentity(t);
369  return;
370  }
371  invdet = 1.0 / det;
372  inv[0] = (float)(t[3] * invdet);
373  inv[2] = (float)(-t[2] * invdet);
374  inv[4] = (float)(((double)t[2] * t[5] - (double)t[3] * t[4]) * invdet);
375  inv[1] = (float)(-t[1] * invdet);
376  inv[3] = (float)(t[0] * invdet);
377  inv[5] = (float)(((double)t[1] * t[4] - (double)t[0] * t[5]) * invdet);
378  }

396  static int glnvg__convertPaint(struct GLNVGcontext* gl, struct GLNVGfragUniforms* frag, struct NVGpaint* paint,
397  struct NVGscissor* scissor, float width, float fringe)
398  {
399  struct GLNVGtexture* tex = NULL;
400  float invxform[6] = {};
401 
402  memset(frag, 0, sizeof(*frag));
403 
404  frag->innerCol = paint->innerColor;
405  frag->outerCol = paint->outerColor;
406 
407  glnvg__xformInverse(invxform, paint->xform);
408  glnvg__xformToMat3x4(frag->paintMat, invxform);
409 
410  if (scissor->extent[0] < 0.5f || scissor->extent[1] < 0.5f)
411  {
412  memset(frag->scissorMat, 0, sizeof(frag->scissorMat));
413  frag->scissorExt[0] = 1.0f;
414  frag->scissorExt[1] = 1.0f;
415  frag->scissorScale[0] = 1.0f;
416  frag->scissorScale[1] = 1.0f;
417  }
418  else
419  {
420  glnvg__xformInverse(invxform, scissor->xform);
421  glnvg__xformToMat3x4(frag->scissorMat, invxform);
422  frag->scissorExt[0] = scissor->extent[0];
423  frag->scissorExt[1] = scissor->extent[1];
424  frag->scissorScale[0] = sqrtf(scissor->xform[0]*scissor->xform[0] + scissor->xform[2]*scissor->xform[2]) / fringe;
425  frag->scissorScale[1] = sqrtf(scissor->xform[1]*scissor->xform[1] + scissor->xform[3]*scissor->xform[3]) / fringe;
426  }
427  memcpy(frag->extent, paint->extent, sizeof(frag->extent));
428  frag->strokeMult = (width*0.5f + fringe*0.5f) / fringe;
429 
430  bgfx::TextureHandle invalid = BGFX_INVALID_HANDLE;
431  gl->th = invalid;
432  if (paint->image != 0)
433  {
434  tex = glnvg__findTexture(gl, paint->image);
435  if (tex == NULL)
436  {
437  return 0;
438  }
439  frag->type = NSVG_SHADER_FILLIMG;
440  frag->texType = tex->type == NVG_TEXTURE_RGBA ? 0.0f : 1.0f;
441  gl->th = tex->id;
442  }
443  else
444  {
445  frag->type = NSVG_SHADER_FILLGRAD;
446  frag->radius = paint->radius;
447  frag->feather = paint->feather;
448  }
449 
450  return 1;
451  }
